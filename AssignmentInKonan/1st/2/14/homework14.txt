void FindFlag(char *dic_1, char *wd_f_1){
    int IsConti = 1;
    int diclen=cont_str(dic_1);
    int wdlen=cont_str(wd_f_1);

    for(int i = 0; i < (diclen-wdlen)+1; i++){
        for(int j = 0; j < wdlen; j++){
            if(IsConti == 0){break;}

            if(*(dic_1+i+j) == *(wd_f_1+j)){
                IsConti = 1;
            }else{
                IsConti = 0;
            }
        }
        //printf("a\n");
        if(IsConti == 1){
            flagL = i;
            flagR = i+wdlen;
            IsFind = 1;
            break;
        }else{
            IsConti = 1;
        }
    }
}


int cont_str(char *c){
    int len=0;
    while(c[len] != '\0'){
        len++;
    }
    return len;
}


char *cat_str(char *dest,const char *src)
{
	char *strDest=dest;
	assert(dest!=NULL && src!=NULL); 

	while(*dest !='\0')                   
	{
		dest++;
	}
	while((*dest++=*src++)!='\0');
    return strDest;
}
